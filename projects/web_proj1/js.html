JavaScript Basics

Make your web pages interactive with JavaScript.

1. Print to Console

console.log("Hello, World!");

2. Variables and Data Types

let name = "John";
let age = 25;

3. Functions in JS

function greet(name) {
    return "Hello, " + name + "!";
}
console.log(greet("John"));

4. Array Manipulation

let fruits = ["Apple", "Banana", "Cherry"];
fruits.push("Mango");

5. Looping through Arrays

for (let i = 0; i < fruits.length; i++) {
    console.log(fruits[i]);
}

6. Event Handling

document.getElementById("btn").addEventListener("click", () => {
    alert("Button Clicked!");
});

7. Manipulating DOM

document.getElementById("demo").innerHTML = "Updated Content!";

8. Conditional Statements

if (age > 18) {
    console.log("You are an adult.");
} else {
    console.log("You are a minor.");
}

9. Working with Objects

let person = {
    name: "Alice",
    age: 28
};
console.log(person.name);

10. Set Timeout Function

setTimeout(() => {
    console.log("This will appear after 2 seconds.");
}, 2000);

11. Array Filter Method

let numbers = [10, 20, 30, 40];
let filtered = numbers.filter(num => num > 20);
console.log(filtered);

12. String Manipulation

let text = "Hello JavaScript!";
console.log(text.toUpperCase());

13. Error Handling

try {
    throw new Error("Something went wrong!");
} catch (error) {
    console.error(error.message);
}

14. Async and Await

async function fetchData() {
    let response = await fetch("https://api.example.com/data");
    let data = await response.json();
    console.log(data);
}
fetchData();

15. Working with Local Storage

localStorage.setItem("username", "JohnDoe");
console.log(localStorage.getItem("username"));
